575     public String getTekstiKentta(int luku) {
576         if (luku == 1) {
577             String teksti = tekstiKentta.getText();
578             tekstiKentta.setText("");
579             return teksti;
580         }
581         if (luku == 2) {
582             String teksti = tekstiKentta1.getText();
583             tekstiKentta.setText("");
584             return teksti;
585         }
586         if (luku == 3) {
587             String teksti = tekstiKentta2.getText();
588             tekstiKentta.setText("");
589             return teksti;
590         }
591         return "";
592     }
