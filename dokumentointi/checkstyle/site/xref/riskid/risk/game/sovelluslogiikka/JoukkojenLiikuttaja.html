3   import riskid.risk.game.domain.*;
4   import riskid.risk.game.kayttoliittyma.GUI;

6   /**
7    * JoukkojenLiikuttaja tekee joukkojen siirtämiseen vaadittavia toimenpiteitä.
8    */
9   public class JoukkojenLiikuttaja {
10  
11      private Taistelusimulaattori ts;
12      private GUI gui;
13  
14      /**
15       * Luokan konstruktori.
16       *
17       * @param gui gui
18       */
19      public JoukkojenLiikuttaja(GUI gui) {
20          this.ts = new Taistelusimulaattori();
21          this.gui = gui;
22      }
23  
24      /**
25       * Metodi joukkojen sijoittamiselle tyhjään ruutuun. Palauttaa true jos
26       * operaatio onnistuu.
27       *
28       * @param minne haluttu alue
29       * @param kenen yksikön pelaaja
30       * @return true/false
31       */
32      public boolean sijoitaAloitusJoukkoja(Alue minne, Pelaaja kenen) {
33          if (minne.getYksikko() == null) {
34              minne.setYksikko(new Yksikko(kenen));
35              minne.setHallitsija(kenen);
36              kenen.poistaReservista(1);
37              return true;
38          } else {
39              gui.uusiAlert("Laiton siirto: sijoita joukkoja tyhjään ruutuun");
40          }
41  
42          return false;
43      }
44  
45      /**
46       * Metodi lisäjoukkojen sijoittamiselle hallittuun ruutuun. Palauttaa true
47       * jos operaatio onnistuu.
48       *
49       * @param minne haluttu alue
50       * @param montako sijoitettava määrä
51       * @param kenen yksikön pelaaja
52       * @return true/false
53       */
54      public boolean sijoitaLisajoukkoja(Alue minne, int montako, Pelaaja kenen) {
55  
56          if (minne.getHallitsija() != null && minne.getHallitsija().equals(kenen)) {
57              minne.getYksikko().setVahvuus(minne.getYksikonVahvuus() + montako);
58              kenen.poistaReservista(montako);
59              return true;
60          } else {
61              gui.uusiAlert("Laiton siirto: lisää joukkoja omille alueillesi");
62          }
63          return false;
64      }
65  
66      /**
67       * Metodi joukkojen siirtämiselle vihollisen hallitsemaan ruutuun.
68       *
69       * @param mista mistä hyökätään
70       * @param mihin haluttu alue
71       * @param montako hyökkääjien vahvuus
72       *
73       */
74      public void hyokkaaTaiValtaa(Alue mista, Alue mihin, int montako) {
75          if (mista.onkoViereinen(mihin)) {
76              if (montako >= 1 && montako < mista.getYksikko().getVahvuus()) {
77                  if (!mista.getHallitsija().equals(mihin.getHallitsija())) {
78                      //taistellaaaaaaaaaaaaaaaan :DDD
79                      liikutaVihollisAlueelle(mista, mihin, montako);
80                  } else {
81                      gui.uusiAlert("Laiton siirto: valtaa vihollisen alueita");
82                  }
83              } else {
84                  gui.uusiAlert("Laiton siirto: liikuta vähintään yhtä ja jätä vähintään yksi taakse");
85              }
86          } else {
87              gui.uusiAlert("Laitoin siirto: alueet eivä ole vierekkäin");
88          }
89      }
90  
91      /**
92       * Metodi joukkojen sijoittamiselle omaan ruutuun. Palauttaa true jos
93       * operaatio onnistuu.
94       *
95       * @param mista mistä siirretän
96       * @param mihin mihin ruutuun
97       * @param montako haluttu määrä
98       * @return true/false
99       */
100     public boolean passiivinenLiike(Alue mista, Alue mihin, int montako) {
101         if (mista.onkoViereinen(mihin)) {
102             if (montako >= 1 && montako < mista.getYksikko().getVahvuus()) {
103                 if (mista.getHallitsija().equals(mihin.getHallitsija())) {
104                     liikutaOmalleAlueelle(mista, mihin, montako);
105                     return true;
106                 } else {
107                     gui.uusiAlert("Laiton siirto: siirrä omalle alueelle");
108                 }
109             } else {
110                 gui.uusiAlert("Laiton siirto: liikuta vähintään yhtä ja jätä vähintään yksi taakse");
111             }
112         } else {
113             gui.uusiAlert("Laitoin siirto: alueet eivä ole vierekkäin");
114         }
115         return false;
116     }
117 
118     private void liikutaOmalleAlueelle(Alue mista, Alue mihin, int montako) {
119         mista.getYksikko().setVahvuus(mista.getYksikonVahvuus() - montako);
120         mihin.getYksikko().setVahvuus(mihin.getYksikonVahvuus() + montako);
121     }
122 
123     private void liikutaVihollisAlueelle(Alue mista, Alue mihin, int montako) {
124         ts.taistele(mista, mihin, montako);
125     }
126 
127 }
