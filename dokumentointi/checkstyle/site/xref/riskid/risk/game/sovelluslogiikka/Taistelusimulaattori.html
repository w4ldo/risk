3   import java.util.*;
4   import riskid.risk.game.domain.*;
5   
6   public class Taistelusimulaattori {
7   
8       private Random random;
9   
10      public Taistelusimulaattori() {
11          this.random = new Random();
12      }
13  
14      //hyökkääjä heittää 1-3 nopppaa, puolustaja 1-2 noppaa.
15      //puolustaja voittaa tasapelit, noppia heitetään kunnes toinen armeija on tuhottu
16      public void taistele(Alue mista, Alue mihin, int montako) {
17          int hyokkaaja = montako;
18          int puolustaja = mihin.getYksikonVahvuus();
19          while (hyokkaaja > 0 && puolustaja > 0) {
20              List<Integer> hyokkaajanHeitot = new ArrayList<>();
21              List<Integer> puolustajanHeitot = new ArrayList<>();
22              if (hyokkaaja >= 3) {
23                  hyokkaajanHeitot.add(this.random.nextInt(6));
24                  hyokkaajanHeitot.add(this.random.nextInt(6));
25                  hyokkaajanHeitot.add(this.random.nextInt(6));
26              } else if (hyokkaaja == 2) {
27                  hyokkaajanHeitot.add(this.random.nextInt(6));
28                  hyokkaajanHeitot.add(this.random.nextInt(6));
29              } else if (hyokkaaja == 1) {
30                  hyokkaajanHeitot.add(this.random.nextInt(6));
31              }
32              if (puolustaja >= 2) {
33                  puolustajanHeitot.add(this.random.nextInt(6));
34                  puolustajanHeitot.add(this.random.nextInt(6));
35              } else if (puolustaja == 1) {
36                  puolustajanHeitot.add(this.random.nextInt(6));
37              }
38              hyokkaajanHeitot.sort(null);
39              puolustajanHeitot.sort(null);
40              if (hyokkaaja > 1 && puolustaja > 1) {
41                  if (hyokkaajanHeitot.get(0) < puolustajanHeitot.get(0)) {
42                      puolustaja--;
43                  } else {
44                      hyokkaaja--;
45                  }
46                  if (hyokkaajanHeitot.get(1) < puolustajanHeitot.get(1)) {
47                      puolustaja--;
48                  } else {
49                      hyokkaaja--;
50                  }
51              } else if (hyokkaajanHeitot.get(0) < puolustajanHeitot.get(0)) {
52                  puolustaja--;
53              } else {
54                  hyokkaaja--;
55              }
56          }
57          if (hyokkaaja > 0) {
58              this.hyokkaajaVoittaa(mista, mihin, montako, hyokkaaja);
59          } else {
60              this.puolustajaVoittaa(mista, mihin, montako, puolustaja);
61          }
62      }
63  
64      private void hyokkaajaVoittaa(Alue mista, Alue mihin, int montako, int hyokkaajat) {
65          mista.getYksikko().setVahvuus(mista.getYksikonVahvuus() - montako);
66          mihin.setYksikko(new Yksikko(mista.getHallitsija()));
67          mihin.getYksikko().setVahvuus(hyokkaajat);
68          mihin.setHallitsija(mista.getHallitsija());
69      }
70  
71      private void puolustajaVoittaa(Alue mista, Alue mihin, int montako, int puolustajat) {
72          mista.getYksikko().setVahvuus(mista.getYksikonVahvuus() - montako);
73          mihin.getYksikko().setVahvuus(puolustajat);
74      }
75  
76  }
