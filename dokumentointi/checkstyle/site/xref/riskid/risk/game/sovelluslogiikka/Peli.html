3   import java.util.Scanner;
4   import riskid.risk.game.domain.*;
5   import riskid.risk.game.kayttoliittyma.GUI;

7   public class Peli {
8   
9       private MapBuilder mb;
10      private JoukkojenLiikuttaja jl;
11      private LuvunKysyja lk;
12      private Kartta map;
13      private Pelaaja pelaajaA;
14      private Pelaaja pelaajaB;
15      private boolean pelaajaAVuoro;
16      private boolean onkoVoittajaa;
17      private GUI gui;
18  
19      public Peli() {
20          this.gui = new GUI();
21          this.mb = new MapBuilder();
22          this.jl = new JoukkojenLiikuttaja(gui);
23          this.map = mb.buildmap();
24          this.pelaajaA = new Pelaaja("Alpha");
25          this.pelaajaB = new Pelaaja("Bravo");
26          this.pelaajaAVuoro = true;
27          this.onkoVoittajaa = false;
28  
29          this.lk = new LuvunKysyja(gui);
30      }
31  
32      public void run() {
33          gui.setVisible(true);
34          gui.teeLisaOsatNakymattomiksi();
35          pelinAlustus();
36          while (!onkoVoittajaa) {
37              if (pelaajaAVuoro) {
38                  uusiVuoro(pelaajaA);
39              } else {
40                  uusiVuoro(pelaajaB);
41              }
42          }
43          gui.paivitaGui(map.tulostaKartta(),
44                  pelaajaAVuoro, pelaajaA.getReservi(),
45                  pelaajaB.getReservi(),
46                  map.laskePelaajanLisajoukot(pelaajaA), map.laskePelaajanLisajoukot(pelaajaB));
47          if (pelaajaAVuoro) {
48              gui.uusiIlmoitus("gg, Alpha voitti");
49          } else {
50              gui.uusiIlmoitus("gg, Bravo voitti");
51          }
52      }
53  
54      private void pelinAlustus() {
55          pelaajaA.lisaaReserviin(57);
56          pelaajaB.lisaaReserviin(57);
57          aloitusjoukkojenSijoitus();
58          aloitusjoukkojenVahvistus();
59      }
60  
61      private void aloitusjoukkojenSijoitus() {
62          gui.uusiIlmoitus("Aloitusjoukkojen sijoitus");
63          while (map.onkoTyhjiaAlueita()) {
64              if (gui.onkoTekstia()) {
65                  int luku = lk.kysyLukua();
66                  if (!lk.onkoLukuSallittu(luku)) {
67                      continue;
68                  }
69                  if (pelaajaAVuoro) {
70                      if (!jl.sijoitaAloitusJoukkoja(map.getAlue(luku), pelaajaA)) {
71                          continue;
72                      }
73                  } else if (!jl.sijoitaAloitusJoukkoja(map.getAlue(luku), pelaajaB)) {
74                      continue;
75                  }
76                  gui.uusiAlert("");
77                  gui.uusiIlmoitus("");
78                  pelaajaAVuoro = !pelaajaAVuoro;
79              }
80              gui.paivitaGui(map.tulostaKartta(),
81                      pelaajaAVuoro, pelaajaA.getReservi(),
82                      pelaajaB.getReservi(),
83                      map.laskePelaajanLisajoukot(pelaajaA), map.laskePelaajanLisajoukot(pelaajaB));
84          }
85      }
86  
87      private void aloitusjoukkojenVahvistus() {
88          gui.uusiIlmoitus("Aloitusjoukkojen vahvistus. Lisää 2x2 yksikköä");
89          while (pelaajaA.getReservi() > 0 || pelaajaB.getReservi() > 0) {
90              if (gui.onkoTekstia()) {
91                  if (this.pelaajaAVuoro) {
92                      lisajoukkojenSijoitus(pelaajaA, 2, 2);
93                  } else {
94                      lisajoukkojenSijoitus(pelaajaB, 2, 2);
95                  }
96                  gui.uusiAlert("");
97                  gui.uusiIlmoitus("");
98                  pelaajaAVuoro = !pelaajaAVuoro;
99              }
100             gui.paivitaGui(map.tulostaKartta(),
101                     pelaajaAVuoro, pelaajaA.getReservi(),
102                     pelaajaB.getReservi(),
103                     map.laskePelaajanLisajoukot(pelaajaA), map.laskePelaajanLisajoukot(pelaajaB));
104         }
105     }
106 
107     private void lisajoukkojenSijoitus(Pelaaja pelaaja, int montakoKertaa, int montakoKerralla) {
108         int i = 0;
109         gui.uusiIlmoitus("Lisäjoukkojen sijoitus");
110         while (i < montakoKertaa) {
111             if (gui.onkoTekstia()) {
112                 int luku = lk.kysyLukua();
113                 if (!lk.onkoLukuSallittu(luku)) {
114                     continue;
115                 }
116                 if (!jl.sijoitaLisajoukkoja(map.getAlue(luku), montakoKerralla, pelaaja)) {
117                     continue;
118                 }
119                 i++;
120             }
121             gui.paivitaGui(map.tulostaKartta(),
122                     pelaajaAVuoro, pelaajaA.getReservi(),
123                     pelaajaB.getReservi(),
124                     map.laskePelaajanLisajoukot(pelaajaA), map.laskePelaajanLisajoukot(pelaajaB));
125         }
126     }
127 
128     private void uusiVuoro(Pelaaja pelaaja) {
129         gui.uusiIlmoitus("Sijoita lisäjoukkoja");
130         pelaaja.lisaaReserviin(map.laskePelaajanLisajoukot(pelaaja));
131         lisajoukkojenSijoitus(pelaaja, map.laskePelaajanLisajoukot(pelaaja) / 3, 3);
132         gui.teeLisaOsatNakyviksi();
133         hyokkaysVaihe(pelaaja);
134         if (onkoVoittajaa) {
135             return;
136         }
137         vahvistusVaihe(pelaaja);
138         gui.teeLisaOsatNakymattomiksi();
139         pelaajaAVuoro = !pelaajaAVuoro;
140     }
141 
142     private void hyokkaysVaihe(Pelaaja pelaaja) {
143         gui.uusiAlert("Hyökkaysvaihe");
144         while (true) {
145             if (gui.onkoTekstia()) {
146                 int mista = lk.kysyLukua();
147                 int mihin = lk.mihinSiirretaan();
148                 int monellako = lk.montakoSiirretaan();
149                 if (lk.onkoLukuSallittu(mista)) {
150                     jl.hyokkaaTaiValtaa(map.getAlue(mista), map.getAlue(mihin), monellako);
151                     if (map.voittaako(pelaaja)) {
152                         onkoVoittajaa = true;
153                         break;
154                     }
155                 } else if (mista == 999) {
156                     break;
157                 }
158             }
159             gui.paivitaGui(map.tulostaKartta(),
160                     pelaajaAVuoro, pelaajaA.getReservi(),
161                     pelaajaB.getReservi(),
162                     map.laskePelaajanLisajoukot(pelaajaA), map.laskePelaajanLisajoukot(pelaajaB));
163         }
164     }
165 
166     private void vahvistusVaihe(Pelaaja pelaaja) {
167         gui.uusiIlmoitus("Vahvistusvaihe, siirrä enintään kolmea omaa joukkoa");
168         int siirtoja = 0;
169         while (siirtoja < 3) {
170             if (gui.onkoTekstia()) {
171                 int mista = lk.kysyLukua();
172                 int mihin = lk.mihinSiirretaan();
173                 int montako = lk.montakoSiirretaan();
174                 if (lk.onkoLukuSallittu(mista)) {
175                     if (jl.passiivinenLiike(map.getAlue(mista), map.getAlue(mihin), montako)) {
176                         siirtoja++;
177                     } else {
178                         gui.uusiAlert("Siirrä vain viereiselle alueelle.");
179                         gui.uusiIlmoitus("Liikuta vähintään yhtä ja jätä vähintään yksi taakse");
180                         continue;
181                     }
182                 } else if (mista == 999) {
183                     break;
184                 } else {
185                     continue;
186                 }
187             }
188             gui.paivitaGui(map.tulostaKartta(),
189                     pelaajaAVuoro, pelaajaA.getReservi(),
190                     pelaajaB.getReservi(),
191                     map.laskePelaajanLisajoukot(pelaajaA), map.laskePelaajanLisajoukot(pelaajaB));
192         }
193     }
